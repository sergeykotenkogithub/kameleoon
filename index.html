Request

Перехват реквеста

Чаще всего используется для отслеживания добавления товара в корзину. Чтобы узнать по какому url идет запрос, нужно открыть вкладку network в devtools 

    
    
(function(open) {
    XMLHttpRequest.prototype.open = function(method, url, async, user, pass) {
         this.addEventListener(
            'readystatechange',
            function() {
                if (this.readyState === 4 && url.includes('add-to-cart')) {
                    console.log('ADD TO CART GOAL');
                }
            },
            false
        );
        open.call(this, method, url, async, user, pass);
    };
})(XMLHttpRequest.prototype.open);
